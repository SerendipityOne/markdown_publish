    区间DP

    详细文档：区间 DP - OI Wiki (oi-wiki.org)

    定义

    区间类动态规划是线性动态规划的扩展，它在分阶段地划分问题时，与阶段中元素出现的顺序和由前一阶段的哪些元素合并而来有很大的关系。

    令状态f(i,j)表示将下标位置 i 到 j 的所有元素合并能获得的价值的最大值，那么f(i,j)=max[f(i,j), f(i,k)+f(k+1,j)+cost]为将这两组元素合并起来的代价。

    性质

    区间 DP 有以下特点:

    1. 合并:即将两个或多个部分进行整合,当然也可以反过来;

    2. 特征:能将问题分解为能两两合并的形式;

    3. 求解：对整个问题设最优值，枚举合并点，将问题分解为左右两个部分，最后合并两个部分的最优值得到原问题的最优值。

    例题

    能量项链：[P1063 NOIP2006 提高组] 能量项链 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)

    核心代码：

    for (int len = 2; len <= n; len++) {       // 枚举每种区间的长度
  for (int i = 1; i <= 2 * n - 1; i++) {  // 枚举区间头的位置，至少留一个出来作为区间尾的位置
     int j = i + len - 1;                 // 区间尾的位置=区间头的位置+区间长度
     // 从区间头开始枚举每个区间的中段，求出每个[i,j]区间的max，直到超出范围
     for (int k = i; k < j && k <= 2 * n - 1; k++) {
        f[i][j] = max(f[i][j], f[i][k] + f[k + 1][j] + tou[i] * wei[k] * wei[j]);
     }
  }
}

    完整代码

    #include <bits/stdc++.h>
using namespace std;
int n;
int tou[210], wei[210];  // 头、尾
// 状态转移方程f(i,j)=max[f(i,j), f(i,k)+f(k+1,j)+cost]
int f[300][300];
int main() {
   cin >> n;
   for (int i = 1; i <= n; i++) {
      cin >> tou[i];
      tou[i + n] = tou[i];  // 处理环，将长度扩展至2n
   }
   for (int i = 1; i <= 2 * n - 1; i++) {
      wei[i] = tou[i + 1];
   }
   wei[2 * n] = tou[1];

   for (int len = 2; len <= n; len++) {       // 枚举每种区间的长度
      for (int i = 1; i <= 2 * n - 1; i++) {  // 枚举区间头的位置，至少留一个出来作为区间尾的位置
         int j = i + len - 1;                 // 区间尾的位置=区间头的位置+区间长度
         // 从区间头开始枚举每个区间的中段，求出每个[i,j]区间的max，直到超出范围
         for (int k = i; k < j && k <= 2 * n - 1; k++) {
            f[i][j] = max(f[i][j], f[i][k] + f[k + 1][j] + tou[i] * wei[k] * wei[j]);
         }
      }
   }
   int ans = -1;
   for (int i = 1; i <= n; i++) {
      // [i,i+n-1]区间的值
      ans = max(ans, f[i][i + n - 1]);
   }
   cout << ans << endl;
}
